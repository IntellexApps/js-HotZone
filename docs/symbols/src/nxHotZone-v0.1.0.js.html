211 		 * Mouse has been pressed, intiate dragging.

216 
217 			// Set where the user has clicked

228 				This.origin.selection = This.selection;

232 		/**
233 		 * Handle the moving of the mouse as hovering or dragging.
234 		 *
235 		 * @param	{object}	event - The event that trigerred this method.
236 		 */
237 		move: function(event) {
238 			var cursor = 'crosshair';
239 			var cursorPosition = null;
240 
241 			// Handle the movement as modification of current selection
242 			if(This.isActive()) {
243 				var x = event.x - This.bounds.left - This.origin.x;
244 				var y = event.y - This.bounds.top  - This.origin.y;
245 
246 				if(This.handler != null) {
247 					var rect = This.handler(This.origin.x, This.origin.y, x, y, This.origin.selection);
248 					This.setSelection(rect);
249 				}
250 
251 			// Do not modify selection, just check the position of the cursor
252 			} else if(This.selection) {
253 				var x = event.x - This.bounds.left - This.selection.left;
254 				var y = event.y - This.bounds.top  - This.selection.top;
255 				var grabZone = This.config.lineGrabZone + This.config.lineWidth;
256 
257 				// Outside of the selection
258 				if(
259 					x <                       - grabZone ||
260 					x > This.selection.width  + grabZone ||
261 					y <                       - grabZone ||
262 					y > This.selection.height + grabZone
263 				) {
264 					cursorPosition = 'out';
265 
266 				// Inside of the selection
267 				} else if(
268 					x >                       + grabZone &&
269 					x < This.selection.width  - grabZone &&
270 					y >                       + grabZone &&
271 					y < This.selection.height - grabZone
272 				) {
273 					cursorPosition = 'in';
274 
275 				// On the line
276 				} else {
277 
278 					// Get angle
279 					var a = Math.round(x - This.selection.width / 2);
280 					var b = Math.round(y - This.selection.height / 2);
281 					var angle = Math.atan2(a, b) / Math.PI * 180 + 180;
282 
283 					// Find direction
284 					cursorPosition = This.getDirection(angle);
285 				}
286 
287 				// Set cursor and handler, based on the current position of the mouse
288 				switch(cursorPosition) {
289 					case 'in':
290 						This.handler = This.handlers.move;
291 						This.canvas.style.cursor = 'move';
292 						break;
293 
294 					case 'out':
295 						This.handler = This.handlers.select;
296 						This.canvas.style.cursor = 'default';
297 						break;
298 
299 					case 'n': case 'nw': case 'w': case 'sw':
300 					case 's': case 'se': case 'e': case 'ne':
301 						This.handler = This.handlers['resize' + cursorPosition.toUpperCase()];
302 						This.canvas.style.cursor = cursorPosition + '-resize';
303 						break;
304 				}
305 			}
306 		},

505 	/**
506 	 * Set and draw the selected zone.
507 	 *
508 	 * @param	{object}	An instance of the rectangle created by nxHotZone.rect.
509 	 */
510 	this.setSelection = function(selection) {
511 		this.image.data = this.loadOriginalData();
512 		this.selection = selection;
513 
514 		// If something is selected
515 		if(this.selection) {
516 
517 			// Set the selection
518 			this.selection = this.selection.restrictToBoundary(0, 0, this.canvas.width, this.canvas.height);
519 
520 			// Precalcualte
521 			var border = {
522 				out: {
523 					l: this.selection.left   - this.config.lineWidth,
524 					r: this.selection.right  + this.config.lineWidth,
525 					t: this.selection.top    - this.config.lineWidth,
526 					b: this.selection.bottom + this.config.lineWidth },
527 				in: {
528 					l: this.selection.left,
529 					r: this.selection.right,
530 					t: this.selection.top,
531 					b: this.selection.bottom }
532 			};
533 
534 			// Process
535 			for(var i = 0, len = this.image.data.length; i < len; i += 4) {
536 
537 				// Get coordinates
538 				var x = i / 4 % this.canvas.width;
539 				var y = Math.floor(i / 4 / this.canvas.width);
540 
541 				// Outside the selection
542 				if(
543 					x < border.out.l ||
544 					x > border.out.r ||
545 					y < border.out.t ||
546 					y > border.out.b ) {
547 
548 					this.pixel(x, y, i, this.config.overlayColor, true);
549 
550 				// Inside the selection
551 				} else if(
552 					x >= border.in.l &&
553 					x <= border.in.r &&
554 					y >= border.in.t &&
555 					y <= border.in.b ) {
556 
557 					this.pixel(x, y, i, this.config.selectedColor);
558 
559 				// Line
560 				} else {
561 					this.pixel(x, y, i, this.config.lineColor);
562 				}
563 			}
564 		} else {
565 			this.handler = this.handlers.select;
566 		}
567 
568 		this.context.putImageData(this.image, 0, 0);
569 	};

571 	/**
572 	 * Handle a color for a single pixel.
573 	 *
574 	 * @param	{int}	x - The x coordinate of the pixel.
575 	 * @param	{int}	y - The x coordinate of the pixel.
576 	 * @param	{mixed}	color - The color to add to the original pixel.
577 	 * @param	{mixed}	blur - The blue radius to use, or 0 to disable (INCOMPLETE).
578 	 */
579 	this.pixel = function(x, y, index, color, blur) {
580 		var pixel = this.getColor(color);
581 
582 		// Set all trhee channels (RGB), with respecto to alpha
583 		for(var i = 0; i < 3; i++) {
584 			this.image.data[index + i] = Math.min(255, Math.round(this.image.data[index + i] * (1 - pixel[3]) + pixel[i] * pixel[3]));
585 		}
586 
587 		// Blur INCOMPLETE
588 		if(blur && this.config.blur) {
589 			var half = this.config.blur / 2;
590 			for(var n = x - half; n < x + half; n++) {
591 				for(var m = y - half; m < y + half; m++) {
592 
593 					// Get index of the pixel to blend with
594 					var b = 4 * (n + m * this.canvas.width);
595 
596 					// Set all trhee channels (RGB)
597 					for(var c = 0; c < 3; c++) {
598 						this.image.data[index + c] = Math.min(255, Math.round((this.image.data[index + c] + this.image.data[b + c]) / 2));
599 					}
600 				}
601 			}
602 		}
603 	};

608 	 * @return	{Uint8ClampedArray}	The image data of the original image.

611 		this.image.data = new Uint8ClampedArray(this.original.data.length);
612 		this.image.data.set(this.original.data);
613 
614 		return this.image.data;
615 	};

618 	 * Get the direction of the resize, based on the angle at which cursor position is looking at the center of selection.
619 	 *
620 	 * @param	{float}	angle - The angle, where 0 is right above the center and increases anti-clockwise.
621 	 */

623 		var directions = [ 'n', 'nw', 'w', 'sw', 's', 'se', 'e', 'ne' ];
624 		var index = 0;

627 			if(angle < i * 45 + 22.5) {
628 				index = i;
629 				break;
630 			}
631 		}
632 
633 		return directions[index];

657 	/**
658 	 * Safe foreach implementation for JSON objects.
659 	 *
660 	 * @param	{Object}	obj	- The object iterate through.
661 	 * @param	{Object}	handler - The handler for each item; receives two parameters, first is the value and the second is the key.
662 	 */
663 	this.foreach = function(obj, handler) {
664 		for(var key in obj) {
665 			if(obj.hasOwnProperty(key)) {
666 				handler.call(this, obj[key], key);
667 			}
668 		}
669 	};

672 	 * Fill in existing JSON object with default values, if those values are missing.
673 	 *
674 	 * @param	{Object}	obj	- The object to fill with default values.
675 	 * @param	{Object}	defaults - The default values to fill in.
676 	 *
677 	 * @returns	{Object}	The obj parameter with additinal keys that do in `defaults` parameter but do not exist in `obj`.

682 				obj[key] = value;

689 	/**
690 	 * Basic rectangle which handle negative widths and heights to create properly oriented rectangle.
691 	 * @class
692 	 *
693 	 * @param	{int}	l - The x coordinate of the left edge, in pixels.
694 	 * @param	{int}	t - The y coordinate of the top edge, in pixels.
695 	 * @param	{int}	w - The desired width of the rectange, in pixels.
696 	 * @param	{int}	h - The desired heigth of the rectange, in pixels.
697 	 *
698 	 * @return	{object}	The resulting rectangle.
699 	 */
700 	this.Rect = function(l, t, w, h) {
701 
702 		// Handle negative width and height
703 		if(w < 0) { l-=w=-w; }
704 		if(h < 0) { t-=h=-h; }
705 
706 		// Set the values
707 		this.left	= l;
708 		this.top	= t;
709 		this.width	= w;
710 		this.height	= h;
711 		this.right	= l + w;
712 		this.bottom	= t + h;
713 
714 		/**
715 		 * Restrict this rectangle to a predefined boundary.
716 		 *
717 		 * @param	{int}	l - The minimal x coordinate of the left edge, in pixels.
718 		 * @param	{int}	t - The minimal y coordinate of the top edge, in pixels.
719 		 * @param	{int}	r - The maximal x coordinate of the right edge, in pixels.
720 		 * @param	{int}	b - The maximal y coordinate of the bottom edge, in pixels.
721 		 *
722 		 * @return	{object}	The restricted rectangle.
723 		 */
724 		this.restrictToBoundary = function(l, t, r, b) {
725 			this.left = Math.min(r - this.width,  Math.max(this.left, l));
726 			this.top  = Math.min(b - this.height, Math.max(this.top,  t));
727 
728 			return this;
729 		}
730 	};

732 	/**
733 	 * A color suitable to use with raw image data.
734 	 * @class
735 	 *
736 	 * @param	{string}	color - A string representation of a color.
737 	 */
738 	this.Color = function(color) {
739 
740 		// Default value is transparent black
741 		this.r = 0;
742 		this.g = 0;
743 		this.b = 0;
744 		this.a = 0;
745 
746 		/**
747 		 * Parse a most common representation of the color: '#AARRGGBB', with optional '#' and 'AA'.
748 		 *
749 		 * @param	{string}	color - A string representation of a color.
750 		 *
751 		 * @return	{nxZone#Color}	The parsed color, or undefined if unable to parse.
752 		 */
753 		this.parseStandard16M = function(color) {
754 			if(match = color.match(/^\s*#?([0-9A-F][0-9A-F])?([0-9A-F][0-9A-F])([0-9A-F][0-9A-F])([0-9A-F][0-9A-F])\s*$/i)) {
755 				return this.load(
756 					parseInt(match[2], 16),
757 					parseInt(match[3], 16),
758 					parseInt(match[4], 16),
759 					typeof match[1] != 'undefined' ? parseInt(match[1], 16) : null
760 				);
761 			}
762 		};
763 
764 		/**
765 		 * Parse a common representation of the color: '#ARGB', with optional '#' and 'A'.
766 		 *
767 		 * @param	{string}	A string representation of a color.
768 		 *
769 		 * @return	{nxZone#Color}	The parsed color, or undefined if unable to parse.
770 		 */
771 		this.parseStandard512 = function(color) {
772 			if(match = color.match(/^\s*#?([0-9A-F])?([0-9A-F])([0-9A-F])([0-9A-F])\s*$/i)) {
773 				return this.load(
774 					parseInt("" + match[2] + match[2], 16),
775 					parseInt("" + match[3] + match[2], 16),
776 					parseInt("" + match[4] + match[2], 16),
777 					typeof match[1] != 'undefined' ? parseInt("" + match[1] + match[1], 16) : null
778 				);
779 			}
780 		};
781 
782 		/**
783 		 * Parse a CSS rbga representation of the color: 'rgba(r, g, b, a)', with optional 'a'.
784 		 *
785 		 * @param	{string}	A string representation of a color.
786 		 *
787 		 * @return	{nxZone#Color}	The parsed color, or null if unable to parse.
788 		 */
789 		this.parseCSSrgba = function(color) {
790 			if(match = color.match(/^\s*rgba?\s*\(\s*([0-9]+)\s*,\s*([0-9]+)\s*,\s*([0-9]+)\s*(,\s*([0-9]+(\.[0-9])?)\s*)?\)\s*$/)) {
791 				return this.load(
792 					parseInt(match[1], 10),
793 					parseInt(match[2], 10),
794 					parseInt(match[3], 10),
795 					typeof match[1] != 'undefined' ? parseFloat(match[1], 10) * 255 : null
796 				);
797 			}
798 		};
799 
800 		/**
801 		 * Helper for loading unchecked values.
802 		 *
803 		 * @param	{int}	r - Red component of the color.
804 		 * @param	{int}	g - Green component of the color.
805 		 * @param	{int}	b - Blue component of the color.
806 		 * @param	{float}	a - Alpha component of the color.
807 		 *
808 		 * @return	{nxZone#Color}	The standardized color.
809 		 */
810 		this.load = function(r, g, b, a) {
811 
812 			// RGB
813 			this.r = Math.min(255, Math.max(0, Math.round(r)));
814 			this.g = Math.min(255, Math.max(0, Math.round(g)));
815 			this.b = Math.min(255, Math.max(0, Math.round(b)));
816 
817 			// Alpha
818 			if(typeof a == 'undefined' || a == null) {
819 				a = 255;
820 			}
821 			this.a = Math.min(255, Math.max(0, Math.round(a)));
822 
823 			return this;
824 		};
825 
826 		// Parse
827 		if(color) {
828 			var methods = [ 'parseStandard512', 'parseStandard16M', 'parseCSSrgba' ];
829 			var length = methods.length;
830 			for(var i = 0; i < length; i++) {
831 				if(this[methods[i]](color) != null) {
832 					break;
833 				}
834 			}
835 		}
836 
837 		/**
838 		 * Get pixel representation of this color.
839 		 *
840 		 * @return	{array}	Array in form of [ R, G, B, A ];
841 		 */
842 		this.toPixel = function() {
843 			return [ this.r, this.g, this.b, this.a ];
844 		};
845 	};
846
